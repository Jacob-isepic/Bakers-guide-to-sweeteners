function calc() { let sugar = Number(document.getElementById("sugar").value); let type = document.getElementById("type").value; if (!sugar || sugar <= 0) return; // LOAD SETTINGS let mode = localStorage.getItem("mode") || "normal"; let adjust = 1; if (mode === "reduced") adjust = 0.85; if (mode === "lowcarb") adjust = 0.7; let message = ""; if (type === "monkfruit") { let blend = sugar * 0.5 * adjust; message = `🧁 MONK FRUIT BLEND Sugar: ${sugar}g Sweetener: ${blend.toFixed(1)}g Mode: ${mode} ✔ Adjusted sweetness applied`; } else if (type === "allulose") { let allulose = sugar * 1.3 * adjust; message = `🍯 ALLULOSE Sugar: ${sugar}g Sweetener: ${allulose.toFixed(1)}g Mode: ${mode} ✔ Texture preserved`; } else if (type === "pure") { let low = (sugar * 1000) / 250; let high = (sugar * 1000) / 150; message = `⚠️ PURE MONK FRUIT EXTRACT Estimated range: ${low.toFixed(1)}mg – ${high.toFixed(1)}mg Mode: ${mode} ✔ Must be diluted before use`; } document.getElementById("result").innerText = message; }
